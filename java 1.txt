const songs = [
    "Song 1 - Artist A",
    "Song 2 - Artist B",
    "Song 3 - Artist C",
    // 將您的歌單加入這裡
];

function selectRandomSong() {
    const randomIndex = Math.floor(Math.random() * songs.length);
    const selectedSong = songs[randomIndex];
    document.getElementById("song-result").textContent = `選中的歌曲是: ${selectedSong}`;
}
